### 7.2.3 各等级详细特征描述

#### L1 初始级（Initial）——"手工作坊"

**典型特征**：
- 建模过程像开盲盒，结果好坏看运气
- 没有标准流程，每个人都有自己的"独门秘籍"
- 数据管理基本靠Excel，版本控制靠"文件名_v1_v2_v3_final_最终版"
- 项目文档？什么是文档？
- 培训新人靠"你坐我旁边看我做"

**工程师吐槽**：
> "我刚来的时候，师傅说'你先看看我怎么做'，看了三个月也没看出个门道。"

**改进建议**：先别想着建体系，把 checklist 做出来，至少保证基础质量。

#### L2 可重复级（Repeatable）——"有了章法"

**典型特征**：
- 有了基本的建模流程，虽然还不够完善
- 开始用 Git 管理代码，但文档还是乱
- 有项目计划，但经常延期
- 能完成项目，但质量波动大
- 开始意识到标准化很重要

**工程师吐槽**：
> "现在至少有流程了，虽然每次都要翻文档找'标准做法'是什么。"

**改进建议**：把流程固化成模板，建立知识库，减少重复劳动。

#### L3 已定义级（Defined）——"步入正轨"

**典型特征**：
- 有完整的标准化流程，新人来了能照着做
- 有独立的质量审核环节
- 知识库开始积累，能复用历史成果
- 项目管理工具用起来，进度可控
- 团队协作顺畅，不再各自为战

**工程师吐槽**：
> "终于不用每个项目都从零开始了，至少能复用30%的内容。"

**改进建议**：开始关注数据驱动，用量化的方式评估质量。

#### L4 量化管理级（Quantitative Management）——"数据说话"

**典型特征**：
- 用数据说话，能说出"我们的模型返工率是12%"
- 质量指标可视化，能及时发现趋势变化
- 有预测能力，能在问题发生前预警
- 持续改进有依据，不再是"我觉得"
- 团队效率可度量，能算出 ROI

**工程师吐槽**：
> "现在老板问'我们做得怎么样'，我能拿出数据，而不是说'还行吧'。"

**改进建议**：开始探索 AI 等新技术，保持行业领先。

#### L5 优化级（Optimizing）——"行业标杆"

**典型特征**：
- 持续优化是团队文化，每个人都在想"怎么做得更好"
- 技术创新是常态，AI 辅助建模已成标配
- 行业影响力，参与标准制定，输出方法论
- 人才吸引力强，成了行业里的"黄埔军校"
- 项目成功率>95%，客户满意度>4.5/5

**工程师吐槽**：
> "在这里工作挺有成就感的，做的东西会被同行认可，技术分享出去有人听。"

**笔者注**：L5 不是终点，是新的起点。达到 L5 的团队也在持续学习，只是他们把学习变成了习惯。

```python
"""
pipe_utils.py - 管网数据处理工具函数库
"""

import pandas as pd
import numpy as np
from scipy import interpolate

def manning_flow(diameter, slope, roughness=0.013):
    """
    使用曼宁公式计算流量
    
    参数:
        diameter: 管径 (mm)
        slope: 坡度 (m/m)
        roughness: 曼宁糙率系数，默认0.013
    
    返回:
        流量 (m³/s)
    """
    D = diameter / 1000  # 转换为米
    A = np.pi * (D/2)**2
    R = D / 4
    Q = (1/roughness) * A * (R**(2/3)) * (slope**0.5)
    return Q

def validate_pipe_data(df):
    """
    验证管网数据质量
    
    参数:
        df: 管段DataFrame，需包含diameter, slope, length列
    
    返回:
        验证结果字典
    """
    issues = []
    
    # 检查空值
    for col in ['diameter', 'slope', 'length']:
        null_count = df[col].isnull().sum()
        if null_count > 0:
            issues.append(f"{col}列有{null_count}个空值")
    
    # 检查管径
    invalid_d = df[(df['diameter'] <= 0) | (df['diameter'] > 5000)]
    if len(invalid_d) > 0:
        issues.append(f"{len(invalid_d)}根管道管径异常")
    
    # 检查坡度
    invalid_s = df[abs(df['slope']) > 0.5]
    if len(invalid_s) > 0:
        issues.append(f"{len(invalid_s)}根管道坡度异常")
    
    return {
        'is_valid': len(issues) == 0,
        'issues': issues,
        'record_count': len(df)
    }

def calculate_catchment_area(nodes, outlet_id, upstream_map):
    """
    计算汇水面积
    
    参数:
        nodes: 节点列表
        outlet_id: 排放口节点ID
        upstream_map: 上游节点映射字典
    
    返回:
        汇水节点列表
    """
    visited = set()
    stack = [outlet_id]
    
    while stack:
        node = stack.pop()
        if node not in visited:
            visited.add(node)
            if node in upstream_map:
                stack.extend(upstream_map[node])
    
    return list(visited)

def interpolate_from_points(known_points, known_values, target_points):
    """
    基于已知点插值
    
    参数:
        known_points: 已知点坐标 [(x1,y1), ...]
        known_values: 已知点值 [v1, v2, ...]
        target_points: 目标点坐标 [(x1,y1), ...]
    
    返回:
        插值结果
    """
    from scipy.interpolate import griddata
    return griddata(known_points, known_values, target_points, method='linear')

# 使用示例
if __name__ == "__main__":
    # 计算流量
    Q = manning_flow(500, 0.005)
    print(f"流量: {Q:.3f} m³/s")
    
    # 验证数据
    sample_data = pd.DataFrame({
        'diameter': [500, 600, 0, 800],
        'slope': [0.005, 0.6, 0.003, 0.004],
        'length': [100, 150, 200, 250]
    })
    result = validate_pipe_data(sample_data)
    print(f"验证结果: {result}")
```

```python
"""
visualization.py - 数据可视化工具
"""

import plotly.express as px
import plotly.graph_objects as go
from plotly.subplots import make_subplots
import pandas as pd

def create_pipe_dashboard(pipe_data):
    """
    创建管网数据仪表盘
    
    参数:
        pipe_data: 管段DataFrame
    
    返回:
        plotly图表
    """
    fig = make_subplots(
        rows=2, cols=2,
        subplot_titles=('管径分布', '坡度分布', '管长分布', '管径-坡度关系'),
        specs=[
            [{"type": "histogram"}, {"type": "box"}],
            [{"type": "histogram"}, {"type": "scatter"}]
        ]
    )
    
    # 管径分布
    fig.add_trace(
        go.Histogram(x=pipe_data['diameter'], name='管径', nbinsx=20),
        row=1, col=1
    )
    
    # 坡度分布
    fig.add_trace(
        go.Box(y=pipe_data['slope'], name='坡度'),
        row=1, col=2
    )
    
    # 管长分布
    fig.add_trace(
        go.Histogram(x=pipe_data['length'], name='管长', nbinsx=20),
        row=2, col=1
    )
    
    # 散点图
    fig.add_trace(
        go.Scatter(
            x=pipe_data['diameter'],
            y=pipe_data['slope'],
            mode='markers',
            name='管径-坡度',
            marker=dict(size=8, opacity=0.6)
        ),
        row=2, col=2
    )
    
    fig.update_layout(height=800, title_text="管网数据仪表盘")
    return fig

def create_flow_hydrograph(flow_data, title="流量过程线"):
    """
    创建流量过程线图
    
    参数:
        flow_data: DataFrame，包含timestamp和flow列
        title: 图表标题
    
    返回:
        plotly图表
    """
    fig = go.Figure()
    
    fig.add_trace(go.Scatter(
        x=flow_data['timestamp'],
        y=flow_data['flow'],
        mode='lines',
        name='流量',
        line=dict(color='blue', width=2)
    ))
    
    # 添加峰值标记
    max_idx = flow_data['flow'].idxmax()
    fig.add_trace(go.Scatter(
        x=[flow_data.loc[max_idx, 'timestamp']],
        y=[flow_data.loc[max_idx, 'flow']],
        mode='markers+text',
        name='峰值',
        text=[f"{flow_data.loc[max_idx, 'flow']:.2f}"],
        textposition='top center',
        marker=dict(color='red', size=10)
    ))
    
    fig.update_layout(
        title=title,
        xaxis_title='时间',
        yaxis_title='流量 (m³/s)',
        hovermode='x unified'
    )
    
    return fig

# 使用示例
if __name__ == "__main__":

    })
    
    # 创建仪表盘
    dashboard = create_pipe_dashboard(pipe_df)
    dashboard.write_html("pipe_dashboard.html")
    print("仪表盘已保存")
```

```python
"""
swmm_input_generator.py - SWMM输入文件生成器
快速生成SWMM的INP文件
"""

def generate_swmm_inp(project_name, junctions, conduits, subcatchments):
    """
    生成SWMM输入文件内容
    
    参数:
        project_name: 项目名称
        junctions: 节点列表 [(name, elev, maxdepth), ...]
        conduits: 管段列表 [(name, inlet, outlet, length, roughness), ...]
        subcatchments: 子汇水区列表 [(name, area, imperv), ...]
    
    返回:
        INP文件内容字符串
    """
    lines = []
    lines.append(f"; SWMM Input File - {project_name}")
    lines.append("; Generated by swmm_input_generator.py")
    lines.append("")
    
    # TITLE section
    lines.append("[TITLE]")
    lines.append(project_name)
    lines.append("")
    
    # OPTIONS section
    lines.append("[OPTIONS]")
    lines.append("FLOW_UNITS CMS")
    lines.append("INFILTRATION HORTON")
    lines.append("FLOW_ROUTING DYNWAVE")
    lines.append("")
    
    # JUNCTIONS section
    lines.append("[JUNCTIONS]")
    lines.append(";;Name Elevation MaxDepth InitDepth SurDepth Aponded")
    for name, elev, maxdepth in junctions:
        lines.append(f"{name} {elev} {maxdepth} 0 0 0")
    lines.append("")
    
    # CONDUITS section
    lines.append("[CONDUITS]")
    lines.append(";;Name Inlet Outlet Length Roughness InOffset OutOffset InitFlow MaxFlow")
    for name, inlet, outlet, length, roughness in conduits:
        lines.append(f"{name} {inlet} {outlet} {length} {roughness} 0 0 0 0")
    lines.append("")
    
    # SUBCATCHMENTS section
    lines.append("[SUBCATCHMENTS]")
    lines.append(";;Name Raingage Outlet Area Imperv Width Slope CurbLength")
    for name, area, imperv in subcatchments:
        width = (area * 10000) ** 0.5  # 简化为正方形
        lines.append(f"{name} RainGage {name}_outlet {area} {imperv} {width:.1f} 0.01 0")
    lines.append("")
    
    return '\n'.join(lines)

# 使用示例
if __name__ == "__main__":
    junctions = [
        ("J1", 10.0, 3.0),
        ("J2", 9.5, 3.0),
        ("J3", 9.0, 3.0),

    conduits = [
        ("C1", "J1", "J2", 100.0, 0.013),
        ("C2", "J2", "J3", 100.0, 0.013),
    ]
    
    subcatchments = [
        ("S1", 1.0, 60),
        ("S2", 1.5, 70),
    ]
    
    inp_content = generate_swmm_inp("示例项目", junctions, conduits, subcatchments)
```

```python
"""
rainfall_processor.py - 降雨数据处理工具
处理各种格式的降雨数据，生成SWMM可用的降雨文件
"""

import pandas as pd
from datetime import datetime

def process_rainfall_station_data(file_path, station_id, interval='1H'):
    """
    处理站点降雨数据
    
    参数:
        file_path: 原始数据文件路径（支持CSV、Excel）
        station_id: 站点ID
        interval: 时间间隔，默认1小时
    
    返回:
        处理后的DataFrame
    """
    # 读取数据
    if file_path.endswith('.csv'):
        df = pd.read_csv(file_path)
    else:
        df = pd.read_excel(file_path)
    
    # 标准列名映射
    column_mapping = {
        '时间': 'datetime',
        'Time': 'datetime',
        '日期': 'datetime',
        'Date': 'datetime',
        '降雨量': 'rainfall',
        'Rainfall': 'rainfall',
        '雨量': 'rainfall',
        'Precipitation': 'rainfall',
    }
    
    # 重命名列
    for old, new in column_mapping.items():
        if old in df.columns:
            df = df.rename(columns={old: new})
    
    # 转换时间格式
    df['datetime'] = pd.to_datetime(df['datetime'])
    df = df.set_index('datetime')
    
    # 重采样到指定间隔
    df_resampled = df.resample(interval).sum()
    
    # 处理缺失值（用0填充，假设无降雨）
    df_resampled['rainfall'] = df_resampled['rainfall'].fillna(0)
    
    # 处理异常值（负值设为0）
    df_resampled['rainfall'] = df_resampled['rainfall'].clip(lower=0)
    
    return df_resampled

def generate_swmm_rainfile(df, station_id, output_path):
    """
    生成SWMM降雨文件
    
    参数:
        df: 降雨DataFrame（包含rainfall列）
        station_id: 站点ID
        output_path: 输出文件路径
    """
    with open(output_path, 'w') as f:
        f.write(f"; Rainfall Data for Station {station_id}\n")
        f.write("; Date Time Rainfall(mm)\n")
        
        for idx, row in df.iterrows():
            date_str = idx.strftime('%Y-%m-%d')
            time_str = idx.strftime('%H:%M')
            rainfall = row['rainfall']
            f.write(f"{date_str} {time_str} {rainfall:.2f}\n")
    
    print(f"降雨文件已生成：{output_path}")
    print(f"总降雨量：{df['rainfall'].sum():.2f} mm")
    print(f"最大雨强：{df['rainfall'].max():.2f} mm/h")

# 使用示例
if __name__ == "__main__":
    # 处理降雨数据
    df = process_rainfall_station_data('原始降雨数据.csv', 'Station001')
    
    # 生成SWMM降雨文件
    generate_swmm_rainfile(df, 'Station001', 'rainfall.dat')
```

```python
"""
validation_report_generator.py - 模型验证报告自动生成
根据观测数据和模拟结果，自动生成验证报告
"""

import pandas as pd
import numpy as np
from scipy import stats

def calculate_nse(observed, simulated):
    """计算纳什效率系数（NSE）"""
    numerator = np.sum((observed - simulated) ** 2)
    denominator = np.sum((observed - np.mean(observed)) ** 2)
    return 1 - numerator / denominator

def calculate_rmse(observed, simulated):
    """计算均方根误差（RMSE）"""
    return np.sqrt(np.mean((observed - simulated) ** 2))

def calculate_mae(observed, simulated):
    """计算平均绝对误差（MAE）"""
    return np.mean(np.abs(observed - simulated))

def calculate_pbias(observed, simulated):
    """计算百分比偏差（PBIAS）"""
    return 100 * np.sum(simulated - observed) / np.sum(observed)

def generate_validation_report(observed_file, simulated_file, output_file):
    """
    生成模型验证报告
    
    参数:
        observed_file: 观测数据文件路径
        simulated_file: 模拟结果文件路径
        output_file: 输出报告文件路径
    """
    # 读取数据
    obs = pd.read_csv(observed_file)
    sim = pd.read_csv(simulated_file)
    
    # 合并数据
    merged = pd.merge(obs, sim, on='datetime', suffixes=('_obs', '_sim'))
    
    # 计算指标
    nse = calculate_nse(merged['flow_obs'], merged['flow_sim'])
    rmse = calculate_rmse(merged['flow_obs'], merged['flow_sim'])
    mae = calculate_mae(merged['flow_obs'], merged['flow_sim'])
    pbias = calculate_pbias(merged['flow_obs'], merged['flow_sim'])
    r2 = stats.pearsonr(merged['flow_obs'], merged['flow_sim'])[0] ** 2
    
    # 评估等级
    if nse > 0.75:
        quality = "优秀"
    elif nse > 0.5:
        quality = "良好"
    elif nse > 0.25:
        quality = "一般"
    else:
        quality = "需改进"
    
    # 生成报告
    report = f"""
# 模型验证报告

## 1. 验证概述
- 观测站点数：{len(obs.columns) - 1}
- 验证时段：{merged['datetime'].min()} 至 {merged['datetime'].max()}
- 数据点数：{len(merged)}

## 2. 验证指标

| 指标 | 数值 | 评价标准 | 结果 |
|------|------|---------|------|
| 纳什效率系数(NSE) | {nse:.3f} | >0.75优秀，>0.5良好 | {quality} |
| 决定系数(R²) | {r2:.3f} | 越接近1越好 | {'合格' if r2 > 0.6 else '需改进'} |
| 均方根误差(RMSE) | {rmse:.3f} m³/s | - | - |
| 平均绝对误差(MAE) | {mae:.3f} m³/s | - | - |
| 百分比偏差(PBIAS) | {pbias:.2f}% | ±10%内优秀 | {'合格' if abs(pbias) < 10 else '需改进'} |

## 3. 结论
模型验证结果为**{quality}**，NSE = {nse:.3f}。

## 4. 建议
"""
    
    if nse < 0.5:
        report += """
- 模型参数需要进一步校准
- 检查输入数据的准确性
- 考虑调整模型结构或边界条件
"""
    else:
        report += """
- 模型可用于后续情景分析
- 建议定期进行验证更新
"""
    
    # 保存报告
    with open(output_file, 'w', encoding='utf-8') as f:
        f.write(report)
    
    print(f"验证报告已生成：{output_file}")
    print(f"NSE = {nse:.3f} ({quality})")

# 使用示例
if __name__ == "__main__":
    generate_validation_report(
        'observed_flow.csv',
        'simulated_flow.csv',
        'validation_report.md'
    )
```
