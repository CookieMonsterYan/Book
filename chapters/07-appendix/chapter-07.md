```python
"""
pipe_utils.py - 管网数据处理工具函数库
"""

import pandas as pd
import numpy as np
from scipy import interpolate

def manning_flow(diameter, slope, roughness=0.013):
    """
    使用曼宁公式计算流量
    
    参数:
        diameter: 管径 (mm)
        slope: 坡度 (m/m)
        roughness: 曼宁糙率系数，默认0.013
    
    返回:
        流量 (m³/s)
    """
    D = diameter / 1000  # 转换为米
    A = np.pi * (D/2)**2
    R = D / 4
    Q = (1/roughness) * A * (R**(2/3)) * (slope**0.5)
    return Q

def validate_pipe_data(df):
    """
    验证管网数据质量
    
    参数:
        df: 管段DataFrame，需包含diameter, slope, length列
    
    返回:
        验证结果字典
    """
    issues = []
    
    # 检查空值
    for col in ['diameter', 'slope', 'length']:
        null_count = df[col].isnull().sum()
        if null_count > 0:
            issues.append(f"{col}列有{null_count}个空值")
    
    # 检查管径
    invalid_d = df[(df['diameter'] <= 0) | (df['diameter'] > 5000)]
    if len(invalid_d) > 0:
        issues.append(f"{len(invalid_d)}根管道管径异常")
    
    # 检查坡度
    invalid_s = df[abs(df['slope']) > 0.5]
    if len(invalid_s) > 0:
        issues.append(f"{len(invalid_s)}根管道坡度异常")
    
    return {
        'is_valid': len(issues) == 0,
        'issues': issues,
        'record_count': len(df)
    }

def calculate_catchment_area(nodes, outlet_id, upstream_map):
    """
    计算汇水面积
    
    参数:
        nodes: 节点列表
        outlet_id: 排放口节点ID
        upstream_map: 上游节点映射字典
    
    返回:
        汇水节点列表
    """
    visited = set()
    stack = [outlet_id]
    
    while stack:
        node = stack.pop()
        if node not in visited:
            visited.add(node)
            if node in upstream_map:
                stack.extend(upstream_map[node])
    
    return list(visited)

def interpolate_from_points(known_points, known_values, target_points):
    """
    基于已知点插值
    
    参数:
        known_points: 已知点坐标 [(x1,y1), ...]
        known_values: 已知点值 [v1, v2, ...]
        target_points: 目标点坐标 [(x1,y1), ...]
    
    返回:
        插值结果
    """
    from scipy.interpolate import griddata
    return griddata(known_points, known_values, target_points, method='linear')

# 使用示例
if __name__ == "__main__":
    # 计算流量
    Q = manning_flow(500, 0.005)
    print(f"流量: {Q:.3f} m³/s")
    
    # 验证数据
    sample_data = pd.DataFrame({
        'diameter': [500, 600, 0, 800],
        'slope': [0.005, 0.6, 0.003, 0.004],
        'length': [100, 150, 200, 250]
    })
    result = validate_pipe_data(sample_data)
    print(f"验证结果: {result}")
```

```python
"""
visualization.py - 数据可视化工具
"""

import plotly.express as px
import plotly.graph_objects as go
from plotly.subplots import make_subplots
import pandas as pd

def create_pipe_dashboard(pipe_data):
    """
    创建管网数据仪表盘
    
    参数:
        pipe_data: 管段DataFrame
    
    返回:
        plotly图表
    """
    fig = make_subplots(
        rows=2, cols=2,
        subplot_titles=('管径分布', '坡度分布', '管长分布', '管径-坡度关系'),
        specs=[
            [{"type": "histogram"}, {"type": "box"}],
            [{"type": "histogram"}, {"type": "scatter"}]
        ]
    )
    
    # 管径分布
    fig.add_trace(
        go.Histogram(x=pipe_data['diameter'], name='管径', nbinsx=20),
        row=1, col=1
    )
    
    # 坡度分布
    fig.add_trace(
        go.Box(y=pipe_data['slope'], name='坡度'),
        row=1, col=2
    )
    
    # 管长分布
    fig.add_trace(
        go.Histogram(x=pipe_data['length'], name='管长', nbinsx=20),
        row=2, col=1
    )
    
    # 散点图
    fig.add_trace(
        go.Scatter(
            x=pipe_data['diameter'],
            y=pipe_data['slope'],
            mode='markers',
            name='管径-坡度',
            marker=dict(size=8, opacity=0.6)
        ),
        row=2, col=2
    )
    
    fig.update_layout(height=800, title_text="管网数据仪表盘")
    return fig

def create_flow_hydrograph(flow_data, title="流量过程线"):
    """
    创建流量过程线图
    
    参数:
        flow_data: DataFrame，包含timestamp和flow列
        title: 图表标题
    
    返回:
        plotly图表
    """
    fig = go.Figure()
    
    fig.add_trace(go.Scatter(
        x=flow_data['timestamp'],
        y=flow_data['flow'],
        mode='lines',
        name='流量',
        line=dict(color='blue', width=2)
    ))
    
    # 添加峰值标记
    max_idx = flow_data['flow'].idxmax()
    fig.add_trace(go.Scatter(
        x=[flow_data.loc[max_idx, 'timestamp']],
        y=[flow_data.loc[max_idx, 'flow']],
        mode='markers+text',
        name='峰值',
        text=[f"{flow_data.loc[max_idx, 'flow']:.2f}"],
        textposition='top center',
        marker=dict(color='red', size=10)
    ))
    
    fig.update_layout(
        title=title,
        xaxis_title='时间',
        yaxis_title='流量 (m³/s)',
        hovermode='x unified'
    )
    
    return fig
```

```python
"""
ai_tools.py - AI工具集成示例
"""

import openai
import json

class HydraulicAIAssistant:
    """水力建模AI助手"""
    
    def __init__(self, api_key):
        openai.api_key = api_key
        self.conversation_history = []
    
    def ask(self, question, context=None):
        """
        向AI助手提问
        
        参数:
            question: 问题
            context: 上下文信息（可选）
        
        返回:
            AI回答
        """
        messages = [
            {"role": "system", "content": "你是水力建模专家助手。"}
        ]
        
        if context:
            messages.append({"role": "user", "content": f"上下文: {context}"})
        
        messages.append({"role": "user", "content": question})
        
        response = openai.ChatCompletion.create(
            model="gpt-4",
            messages=messages,
            temperature=0.3
        )
        
        answer = response.choices[0].message.content
        
        # 记录对话
        self.conversation_history.append({
            'question': question,
            'answer': answer
        })
        
        return answer
    
    def generate_code(self, requirements):
        """
        生成Python代码
        
        参数:
            requirements: 代码需求描述
        
        返回:
            生成的代码
        """
        prompt = f"""
        请根据以下需求生成Python代码：
        
        {requirements}
        
        要求：
        1. 使用pandas处理数据
        2. 添加详细注释
        3. 包含输入验证
        4. 返回完整的可运行代码
        """
        
        response = openai.ChatCompletion.create(
            model="gpt-4",
            messages=[
                {"role": "system", "content": "你是Python数据工程专家。"},
                {"role": "user", "content": prompt}
            ],
            temperature=0.2
        )
        
        return response.choices[0].message.content
    
    def analyze_results(self, results_summary):
        """
        分析模型结果
        
        参数:
            results_summary: 结果摘要
        
        返回:
            分析结论
        """
        prompt = f"""
        请分析以下水力模型结果：
        
        {json.dumps(results_summary, ensure_ascii=False)}
        
        请提供：
        1. 关键发现
        2. 存在的问题
        3. 改进建议
        """
        
        return self.ask(prompt)
```
